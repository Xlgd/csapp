long fact_for(long n) {
    long i;
    long result = 1;
    for (i = 2; i <= n; i++){
        result *= i;
    }
    return result;
}

long fact_for_while(long n) {
    long i;
    long result = 1;
    while (i <= n) {
        result *= i;
        i++;
    }
    return result;
}

long fact_for_guarded_do(long n) {
    long i;
    long result;
    if (i > n) {
        goto done;
    }
loop:
    result *= i;
    i++;
    if (i <= n) {
        goto loop;
    }
done:
    return result;
}
